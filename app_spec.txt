<project_specification>
  <project_name>ArchCanvas</project_name>

  <overview>
    ArchCanvas is a visual architecture design tool where software systems are represented as typed,
    composable, code-bound graph nodes on an interactive canvas. The .archc binary file (Protocol Buffers)
    is the single source of truth, and ALL access flows through three APIs: Render, Text, and Export.
    AI and humans interact through the same Text API. The canvas and codebase evolve together.
  </overview>

  <storage_architecture>
    <description>
      ArchCanvas uses a BINARY FILE format (.archc) as the single source of truth.
      There is NO server-side database. All data is serialized via Protocol Buffers
      and stored in .archc files on the user's local filesystem.
      The file format includes magic bytes ("ARCHC\x00"), a format version, and a SHA-256
      checksum for integrity verification.
    </description>
    <file_format>
      - Magic bytes: "ARCHC\x00" + uint16 format version (big-endian)
      - Payload: Protocol Buffer encoded ArchCanvasFile message
      - Contains: FileHeader, Architecture (graph), CanvasState, AIState, UndoHistory
    </file_format>
    <persistence>
      - Primary: File System Access API (Chrome) for save-in-place
      - Fallback: File/Blob download for non-Chrome browsers
      - Sidecar: Auto-generated .summary.md markdown file alongside .archc
    </persistence>
  </storage_architecture>

  <protobuf_schema>
    <messages>
      - ArchCanvasFile: top-level container (header, architecture, canvas_state, ai_state, undo_history)
      - FileHeader: format_version, tool_version, timestamps, checksum_sha256
      - Architecture: name, description, owners, nodes[], edges[]
      - Node: id, type, display_name, args (map), code_refs[], notes[], properties (map), position, children[] (recursive), ref_source
      - Edge: id, from_node, to_node, from_port, to_port, type (enum: SYNC/ASYNC/DATA_FLOW), label, properties, notes[]
      - Note: id, author, timestamp_ms, content (markdown), tags[], status (enum: NONE/PENDING/ACCEPTED/DISMISSED), suggestion_type
      - CodeRef: path, role (enum: SOURCE/API_SPEC/SCHEMA/DEPLOYMENT/CONFIG/TEST)
      - Position: x, y, width, height, color
      - CanvasState: viewport (x, y, zoom), selected_node_ids[], navigation_path[], panel_layout
      - PanelLayout: right_panel_open, right_panel_tab, right_panel_width
      - AIState: conversations[]
      - AIConversation: id, scoped_to_node_id, messages[], created_at_ms
      - AIMessage: id, role, content, timestamp_ms, suggestions[]
      - AISuggestion: id, target_node_id, target_edge_id, suggestion_type, content, status
      - UndoHistory: entries[], current_index, max_entries
      - UndoEntry: description, timestamp_ms, architecture_snapshot (bytes)
      - Value: oneof (string, number, bool, string_list)
    </messages>
  </protobuf_schema>

  <core_features>
    <binary_core_engine>
      - Protocol Buffer schema compilation and TypeScript codegen
      - Binary codec: encode/decode .archc files with magic bytes and checksum
      - Proto-to-TypeScript type converters (bidirectional)
      - File I/O: open, save, save-as using File System Access API with fallback
      - File format version migration support
      - Integrity verification via SHA-256 checksum
    </binary_core_engine>

    <graph_engine>
      - Add/remove/update nodes (pure functions, immutable)
      - Add/remove/update edges
      - Add/remove/update notes on nodes and edges
      - Add child nodes (recursive nesting for fractal zoom)
      - Move node positions
      - Find node/edge by ID (recursive search through children)
      - Find node parent
      - Get node path (navigation breadcrumb path)
    </graph_engine>

    <graph_query_and_traversal>
      - Get nodes at navigation level (path-based: [] = root, ["node-id"] = children)
      - Get edges at navigation level (filter to current visible nodes)
      - Get external edges (boundary edges crossing navigation levels)
      - Get neighbors within N hops
      - Full-text search across node names, properties, notes, edge labels
    </graph_query_and_traversal>

    <undo_redo>
      - Snapshot-based undo/redo stack
      - Max 100 entries (configurable)
      - Branch behavior: new action after undo discards redo future
      - Serialized Architecture snapshots for compact storage
      - Undo/redo state persisted in .archc file
    </undo_redo>

    <nodedef_registry>
      - 15 built-in nodedef YAML files organized by namespace
      - Zod schema validation for nodedef structure
      - Registry manager: load, resolve by type, list by namespace, search
      - Build-time YAML import via Vite
      - Nodedefs organized into categories:
        - Compute: service, function, worker, api-gateway
        - Data: database, cache, object-storage, repository
        - Messaging: message-queue, event-bus, stream-processor
        - Network: load-balancer, cdn, dns
        - Observability: logging, monitoring
    </nodedef_registry>

    <text_api>
      - describe(): Full architecture description in ai/human/structured formats
      - listNodes(): Node summaries with child/note/connection counts
      - getNode(): Detailed node info with args, properties, connections, notes
      - getEdges(): Edge summaries
      - getNodeDef(): Resolve nodedef type info
      - search(): Full-text search across graph
      - addNode/addEdge/addNote/updateNode/removeNode/removeEdge mutations
      - suggest(): AI creates pending note suggestions
      - resolveSuggestion(): Accept or dismiss AI suggestions
      - getAIContext(): Convenience method for AI context with neighbors
    </text_api>

    <formatters>
      - AI Formatter: XML-structured output optimized for LLM token efficiency
      - Human Formatter: Markdown output for terminal/human readability with emoji badges
      - Structured Formatter: Clean JSON output for programmatic consumption
    </formatters>

    <render_api>
      - Transform internal ArchGraph to React Flow nodes/edges
      - Navigation-level-aware rendering (fractal zoom)
      - NodeDef type to React component mapping
      - Port handle generation from nodedef spec
      - Badge counts (notes, suggestions, code refs)
    </render_api>

    <export_api>
      - Markdown summary sidecar (auto-generated on save)
      - Mermaid diagram generation with proper shape mapping
      - PNG export via html-to-image
      - SVG export via html-to-image
    </export_api>
