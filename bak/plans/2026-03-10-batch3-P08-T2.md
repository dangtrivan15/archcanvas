### FileSystemAccessBackend

Wraps Chrome's File System Access API (currently in `webFileSystemAdapter.ts`):

```typescript
export class FileSystemAccessBackend implements StorageBackend {
  readonly type = 'file-system-access';
  readonly capabilities = { supportsDirectWrite: true, supportsLastModified: true };

  async openFilePicker(): Promise<StorageHandle | null> {
    // Use window.showOpenFilePicker() — same logic as webFileSystemAdapter.pickFileViaFSA()
    // Return StorageHandle with _internal: FileSystemFileHandle
  }

  async read(handle: StorageHandle): Promise<Uint8Array> {
    const fsHandle = handle._internal as FileSystemFileHandle;
    const file = await fsHandle.getFile();
    return new Uint8Array(await file.arrayBuffer());
  }

  async write(handle: StorageHandle, data: Uint8Array): Promise<StorageHandle> {
    const fsHandle = handle._internal as FileSystemFileHandle;
    const writable = await fsHandle.createWritable();
    await writable.write(data);
    await writable.close();
    return handle;
  }

  async saveFilePicker(data, options): Promise<StorageHandle | null> {
    // Use window.showSaveFilePicker() — same logic as webFileSystemAdapter.saveFileAsViaFSA()
  }
}
```

### Backend Selection

Add a factory function to select the right backend:

```typescript
// In storageManager.ts or a new backends/index.ts
export function createDefaultBackend(): StorageBackend {
  if (typeof window !== 'undefined' && 'showOpenFilePicker' in window) {
    return new FileSystemAccessBackend();
  }
  return new FileDownloadBackend();
}
```

### Relationship to FileSystemAdapter

The new backends **do not replace** `fileSystemAdapter.ts`. That remains the P06
platform abstraction layer. The backends are a higher-level abstraction:

- `FileSystemAccessBackend` uses browser APIs directly (same as `webFileSystemAdapter`)
- `FileDownloadBackend` uses browser APIs directly (same as `webFileSystemAdapter` fallback)
- Future: a `NativeBackend` could wrap `nativeFileSystemAdapter`

The key difference: `FileSystemAdapter` returns raw result types, while `StorageBackend`
returns `StorageHandle` — a typed, backend-aware handle.
